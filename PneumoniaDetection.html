Vectorization Image Data and Its Conversion to Tensorflow Objects¶

In [ ]:
# Now we have all training, validation image paths and their respective labels 
BATCH_SIZE = 32
IMAGE_SIZE = 224

# Function used for Transformation (vectorization of images)
def load_and_transform(image , label , train = True):
    image = tf.io.read_file(image)
    image = tf.io.decode_jpeg(image, channels = 3)
    image = tf.image.resize(image, [IMAGE_SIZE, IMAGE_SIZE], method = "nearest")
    if train:
        image = tf.image.random_flip_left_right(image)
    return image, label

# Function used to Create a Tensorflow Data Object
def get_dataset(paths, labels , train = True):
    # Converting images and labes into tensorflow objects
    image_paths = tf.convert_to_tensor(paths)
    labels = tf.convert_to_tensor(labels)

    # Converting tf objects into tf datasets
    image_dataset = tf.data.Dataset.from_tensor_slices(image_paths)
    label_dataset = tf.data.Dataset.from_tensor_slices(labels)

    # Putting (zipping) two datasets together
    dataset = tf.data.Dataset.zip((image_dataset, label_dataset)).shuffle(500)

    # Loading and transforming (quantifying) images 
    dataset = dataset.map(lambda image, label: load_and_transform(image , label, train))
    dataset = dataset.repeat()
    dataset = dataset.shuffle(1000)
    dataset = dataset.batch(BATCH_SIZE)

    return dataset

# Building ResNet50 model
from tensorflow.keras.applications import ResNet50V2

backbone = ResNet50V2(
    input_shape = (IMAGE_SIZE, IMAGE_SIZE, 3),
    include_top = False
)

model = tf.keras.Sequential([
    backbone,
    tf.keras.layers.BatchNormalization(),
    tf.keras.layers.GlobalAveragePooling2D(),
    tf.keras.layers.Dropout(0.25),
    tf.keras.layers.BatchNormalization(),
    tf.keras.layers.Dense(64, activation = 'relu'),
    tf.keras.layers.Dense(1, activation = 'sigmoid')
])
